class ServerMetrics:
    """Thread-safe metrics collector."""

    def __init__(self) -> None:
        self._lock = Lock()
        self.requests_total = 0
        self.requests_active = 0
        self.requests_succeeded = 0
        self.requests_failed = 0
        self.total_response_time_ms = 0.0
        self.tokens_prompt = 0
        self.tokens_completion = 0
        self.start_time = time.time()

    def record_request_start(self) -> None:
        with self._lock:
            self.requests_total += 1
            self.requests_active += 1

    def record_request_end(self, success: bool, duration_ms: float) -> None:
        with self._lock:
            self.requests_active -= 1
            self.total_response_time_ms += duration_ms
            if success:
                self.requests_succeeded += 1
            else:
                self.requests_failed += 1

    def record_tokens(self, prompt: int, completion: int) -> None:
        with self._lock:
            self.tokens_prompt += prompt
            self.tokens_completion += completion

    def get_metrics(self) -> MetricsResponse:
        with self._lock:
            completed = self.requests_succeeded + self.requests_failed
            avg_time = self.total_response_time_ms / completed if completed > 0 else 0.0
            return MetricsResponse(
                requests_total=self.requests_total,
                requests_active=self.requests_active,
                requests_succeeded=self.requests_succeeded,
                requests_failed=self.requests_failed,
                avg_response_time_ms=avg_time,
                uptime_seconds=time.time() - self.start_time,
                tokens_used={
                    "prompt": self.tokens_prompt,
                    "completion": self.tokens_completion,
                    "total": self.tokens_prompt + self.tokens_completion,
                },
            )


class AgentAPIServer:
    """Production-ready API server for BeeAI agents."""

    def __init__(
        self,
        agent: AnyAgent,
        config: ModelLike[DeployConfig] | None = None,
        memory_manager: MemoryManager | None = None,
    ) -> None:
        self._agent = agent
        self._config = to_model(DeployConfig, config or DeployConfig())
        self._memory_manager = memory_manager or UnlimitedMemoryManager()
        self._metrics = ServerMetrics()
        self._ready = False
        self._semaphore: asyncio.Semaphore | None = None
        self._app: FastAPI | None = None

    @cached_property
    def app(self) -> "FastAPI":
        """Get or create the FastAPI application."""
        return self._create_app()

    def _create_app(self) -> "FastAPI":
        """Create the FastAPI application with all endpoints."""
        try:
            from fastapi import FastAPI, Header, HTTPException, Request, status
            from fastapi.middleware.cors import CORSMiddleware
            from fastapi.responses import JSONResponse
            from sse_starlette.sse import EventSourceResponse
        except ModuleNotFoundError as e:
            raise DeploymentError(
                "Required modules not found. Run 'pip install \"beeai-framework[deploy]\"' to install.",
                cause=e,
            )

        @asynccontextmanager
        async def lifespan(app: FastAPI) -> AsyncGenerator[None, None]:
            self._semaphore = asyncio.Semaphore(self._config.max_concurrent_requests)
            self._ready = True
            logger.info(f"Server started on {self._config.host}:{self._config.port}")
            yield
            self._ready = False
            logger.info("Server shutting down")

        fastapi_kwargs: dict[str, Any] = {
            "title": self._config.title,
            "description": self._config.description,
            "version": self._config.version,
            "lifespan": lifespan,
        }
        if not self._config.enable_docs:
            fastapi_kwargs["docs_url"] = None
            fastapi_kwargs["redoc_url"] = None
            fastapi_kwargs["openapi_url"] = None

        fastapi_kwargs.update(self._config.fastapi_kwargs)

        app = FastAPI(**fastapi_kwargs)

        app.add_middleware(
            CORSMiddleware,
            allow_origins=self._config.cors_origins,
            allow_credentials=True,
            allow_methods=["*"],
            allow_headers=["*"],
        )

        def _validate_api_key(api_key: str | None) -> None:
            if self._config.api_key is None:
                return
            if api_key is None:
                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Missing API key")
            token = api_key.replace("Bearer ", "") if api_key.startswith("Bearer ") else api_key
            if token != self._config.api_key:
                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid API key")

        @app.post("/run", response_model=RunResponse, responses={400: {"model": ErrorResponse}})
        async def run_endpoint(
            request: RunRequest,
            authorization: str | None = Header(None, alias="Authorization"),
        ) -> JSONResponse:
            """Execute the agent and return the response."""
            _validate_api_key(authorization)
            request_id = str(uuid.uuid4())
            start_time = time.time()
            self._metrics.record_request_start()

            try:
                if self._semaphore is None:
                    raise HTTPException(status_code=status.HTTP_503_SERVICE_UNAVAILABLE, detail="Server not ready")

                async with self._semaphore:
                    result = await self._run_agent(request)
                    duration_ms = (time.time() - start_time) * 1000
                    self._metrics.record_request_end(True, duration_ms)

                    if result.usage:
                        self._metrics.record_tokens(
                            result.usage.get("prompt_tokens", 0),
                            result.usage.get("completion_tokens", 0),
                        )

                    return JSONResponse(content=result.model_dump())

            except HTTPException:
                raise
            except ValidationError as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(False, duration_ms)
                return JSONResponse(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    content=ErrorResponse(
                        error="Invalid request",
                        code="VALIDATION_ERROR",
                        details={"errors": e.errors()},
                        request_id=request_id,
                    ).model_dump(),
                )
            except Exception as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(False, duration_ms)
                logger.error(f"Agent execution failed: {e}")
                return JSONResponse(
                    status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                    content=ErrorResponse(
                        error=str(e),
                        code="AGENT_EXECUTION_ERROR",
                        request_id=request_id,
                    ).model_dump(),
                )

        @app.post("/stream")
        async def stream_endpoint(
            request: RunRequest,
            authorization: str | None = Header(None, alias="Authorization"),
        ) -> EventSourceResponse:
            """Execute the agent and stream the response."""
            _validate_api_key(authorization)
            self._metrics.record_request_start()
            start_time = time.time()

            async def event_generator() -> AsyncGenerator[dict[str, Any], None]:
                try:
                    if self._semaphore is None:
                        yield {
                            "event": StreamEventType.ERROR.value,
                            "data": to_json(StreamEvent(type=StreamEventType.ERROR, data="Server not ready").model_dump()),
                        }
                        return

                    async with self._semaphore:
                        yield {
                            "event": StreamEventType.START.value,
                            "data": to_json(StreamEvent(type=StreamEventType.START).model_dump()),
                        }

                        async for event in self._stream_agent(request):
                            yield {
                                "event": event.type.value,
                                "data": to_json(event.model_dump()),
                            }

                        duration_ms = (time.time() - start_time) * 1000
                        self._metrics.record_request_end(True, duration_ms)

                except Exception as e:
                    duration_ms = (time.time() - start_time) * 1000
                    self._metrics.record_request_end(False, duration_ms)
                    logger.error(f"Streaming failed: {e}")
                    yield {
                        "event": StreamEventType.ERROR.value,
                        "data": to_json(StreamEvent(type=StreamEventType.ERROR, data=str(e)).model_dump()),
                    }

            return EventSourceResponse(event_generator())

        @app.get("/health", response_model=HealthResponse)
        async def health_endpoint() -> JSONResponse:
            """Health check endpoint."""
            checks = {
                "agent_loaded": self._agent is not None,
                "server_ready": self._ready,
            }
            all_healthy = all(checks.values())

            return JSONResponse(
                status_code=status.HTTP_200_OK if all_healthy else status.HTTP_503_SERVICE_UNAVAILABLE,
                content=HealthResponse(
                    status=HealthStatus.HEALTHY if all_healthy else HealthStatus.UNHEALTHY,
                    version=self._config.version,
                    uptime_seconds=time.time() - self._metrics.start_time,
                    checks=checks,
                ).model_dump(),
            )

        if self._config.enable_metrics:
            @app.get("/metrics", response_model=MetricsResponse)
            async def metrics_endpoint(
                authorization: str | None = Header(None, alias="Authorization"),
            ) -> JSONResponse:
                """Metrics endpoint."""
                _validate_api_key(authorization)
                return JSONResponse(content=self._metrics.get_metrics().model_dump())

        return app

    async def _get_memory(self, session_id: str | None) -> BaseMemory:
        """Get or create memory for the session."""
        if session_id is None:
            memory = await self._agent.memory.clone()
            memory.reset()
            return memory

        if not await self._memory_manager.contains(session_id):
            memory = await self._agent.memory.clone()
            memory.reset()
            await self._memory_manager.set(session_id, memory)

        return await self._memory_manager.get(session_id)

    def _prepare_messages(self, request: RunRequest) -> list[UserMessage | AssistantMessage | SystemMessage]:
        """Convert request input to messages."""
        if isinstance(request.input, str):
            return [UserMessage(content=request.input)]

        messages: list[UserMessage | AssistantMessage | SystemMessage] = []
        for msg in request.input:
            if msg.role == MessageRole.USER:
                messages.append(UserMessage(content=msg.content))
            elif msg.role == MessageRole.ASSISTANT:
                messages.append(AssistantMessage(content=msg.content))
            elif msg.role == MessageRole.SYSTEM:
                messages.append(SystemMessage(content=msg.content))
        return messages

    async def _run_agent(self, request: RunRequest) -> RunResponse:
        """Execute the agent and return the response."""
        messages = self._prepare_messages(request)

        run_kwargs: dict[str, Any] = {}
        if request.context:
            run_kwargs["context"] = request.context
        if request.max_iterations is not None:
            run_kwargs["max_iterations"] = request.max_iterations

        original_memory = self._agent.memory
        try:
            self._agent.memory = await self._get_memory(request.session_id)
            result = await self._agent.run(messages, **run_kwargs)

            usage = None
            if hasattr(result, "usage") and result.usage:
                usage = {
                    "prompt_tokens": result.usage.prompt_tokens,
                    "completion_tokens": result.usage.completion_tokens,
                    "total_tokens": result.usage.total_tokens,
                }

            response_messages = [
                MessageContent(
                    role=MessageRole.ASSISTANT if hasattr(msg, "role") and msg.role == "assistant" else MessageRole.USER,
                    content=msg.text,
                )
                for msg in result.output
            ]

            return RunResponse(
                output=result.last_message.text,
                messages=response_messages,
                session_id=request.session_id,
                usage=usage,
                metadata={
                    "iterations": len(result.iterations) if hasattr(result, "iterations") else None,
                },
            )
        except Exception as e:
            raise AgentExecutionError(str(e), cause=e)
        finally:
            self._agent.memory = original_memory

    async def _stream_agent(self, request: RunRequest) -> AsyncGenerator[StreamEvent, None]:
        """Execute the agent and stream events."""
        messages = self._prepare_messages(request)

        run_kwargs: dict[str, Any] = {}
        if request.context:
            run_kwargs["context"] = request.context
        if request.max_iterations is not None:
            run_kwargs["max_iterations"] = request.max_iterations

        original_memory = self._agent.memory
        try:
            self._agent.memory = await self._get_memory(request.session_id)
            run = self._agent.run(messages, **run_kwargs)

            async for data, meta in run:
                event = self._convert_event(data, meta)
                if event:
                    yield event

            result = await run
            yield StreamEvent(
                type=StreamEventType.DONE,
                data={
                    "output": result.last_message.text,
                    "session_id": request.session_id,
                },
            )

        except Exception as e:
            yield StreamEvent(type=StreamEventType.ERROR, data=str(e))
            raise
        finally:
            self._agent.memory = original_memory

    def _convert_event(self, data: Any, meta: EventMeta) -> StreamEvent | None:
        """Convert agent event to stream event."""
        if meta.name == "update" and hasattr(data, "update"):
            update = data.update
            if hasattr(update, "key"):
                if update.key == "thought":
                    return StreamEvent(type=StreamEventType.THINKING, data=str(update.value))
                elif update.key == "tool_name":
                    return StreamEvent(
                        type=StreamEventType.TOOL_CALL,
                        data={"name": str(update.value)},
                    )
                elif update.key == "tool_output":
                    return StreamEvent(
                        type=StreamEventType.TOOL_RESULT,
                        data={"output": str(update.value)},
                    )
                elif update.key == "final_answer":
                    return StreamEvent(type=StreamEventType.TOKEN, data=str(update.value))
        elif meta.name == "partial_update" and hasattr(data, "update"):
            update = data.update
            if hasattr(update, "key") and update.key == "final_answer":
                return StreamEvent(type=StreamEventType.TOKEN, data=str(update.value))

        return None

    def serve(self, **uvicorn_kwargs: Any) -> None:
        """Start the server (blocking)."""
        try:
            import uvicorn
        except ModuleNotFoundError as e:
            raise ServerStartError(
                "uvicorn not found. Run 'pip install \"beeai-framework[deploy]\"' to install.",
                cause=e,
            )

        kwargs: dict[str, Any] = {
            "host": self._config.host,
            "port": self._config.port,
        }
        kwargs.update(uvicorn_kwargs)

        uvicorn.run(self.app, **kwargs)

class ServerMetrics:
    """Thread-safe metrics collector."""

    def __init__(self) -> None:
        self._lock = Lock()
        self.requests_total = 0
        self.requests_active = 0
        self.requests_succeeded = 0
        self.requests_failed = 0
        self.total_response_time_ms = 0.0
        self.tokens_prompt = 0
        self.tokens_completion = 0
        self.start_time = time.time()

    def record_request_start(self) -> None:
        with self._lock:
            self.requests_total += 1
            self.requests_active += 1

    def record_request_end(self, success: bool, duration_ms: float) -> None:
        with self._lock:
            self.requests_active -= 1
            self.total_response_time_ms += duration_ms
            if success:
                self.requests_succeeded += 1
            else:
                self.requests_failed += 1

    def record_tokens(self, prompt: int, completion: int) -> None:
        with self._lock:
            self.tokens_prompt += prompt
            self.tokens_completion += completion

    def get_metrics(self) -> MetricsResponse:
        with self._lock:
            completed = self.requests_succeeded + self.requests_failed
            avg_time = self.total_response_time_ms / completed if completed > 0 else 0.0
            return MetricsResponse(
                requests_total=self.requests_total,
                requests_active=self.requests_active,
                requests_succeeded=self.requests_succeeded,
                requests_failed=self.requests_failed,
                avg_response_time_ms=avg_time,
                uptime_seconds=time.time() - self.start_time,
                tokens_used={
                    "prompt": self.tokens_prompt,
                    "completion": self.tokens_completion,
                    "total": self.tokens_prompt + self.tokens_completion,
                },
            )


class AgentAPIServer:
    """Production-ready API server for BeeAI agents."""

    def __init__(
        self,
        agent: AnyAgent,
        config: ModelLike[DeployConfig] | None = None,
        memory_manager: MemoryManager | None = None,
    ) -> None:
        self._agent = agent
        self._config = to_model(DeployConfig, config or DeployConfig())
        self._memory_manager = memory_manager or UnlimitedMemoryManager()
        self._metrics = ServerMetrics()
        self._ready = False
        self._semaphore: asyncio.Semaphore | None = None
        self._app: FastAPI | None = None

    @cached_property
    def app(self) -> "FastAPI":
        """Get or create the FastAPI application."""
        return self._create_app()

    def _create_app(self) -> "FastAPI":
        """Create the FastAPI application with all endpoints."""
        try:
            from fastapi import FastAPI, Header, HTTPException, Request, status
            from fastapi.middleware.cors import CORSMiddleware
            from fastapi.responses import JSONResponse
            from sse_starlette.sse import EventSourceResponse
        except ModuleNotFoundError as e:
            raise DeploymentError(
                "Required modules not found. Run 'pip install \"beeai-framework[deploy]\"' to install.",
                cause=e,
            )

        @asynccontextmanager
        async def lifespan(app: FastAPI) -> AsyncGenerator[None, None]:
            self._semaphore = asyncio.Semaphore(self._config.max_concurrent_requests)
            self._ready = True
            logger.info(f"Server started on {self._config.host}:{self._config.port}")
            yield
            self._ready = False
            logger.info("Server shutting down")

        fastapi_kwargs: dict[str, Any] = {
            "title": self._config.title,
            "description": self._config.description,
            "version": self._config.version,
            "lifespan": lifespan,
        }
        if not self._config.enable_docs:
            fastapi_kwargs["docs_url"] = None
            fastapi_kwargs["redoc_url"] = None
            fastapi_kwargs["openapi_url"] = None

        fastapi_kwargs.update(self._config.fastapi_kwargs)

        app = FastAPI(**fastapi_kwargs)

        app.add_middleware(
            CORSMiddleware,
            allow_origins=self._config.cors_origins,
            allow_credentials=True,
            allow_methods=["*"],
            allow_headers=["*"],
        )

        def _validate_api_key(api_key: str | None) -> None:
            if self._config.api_key is None:
                return
            if api_key is None:
                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Missing API key")
            token = api_key.replace("Bearer ", "") if api_key.startswith("Bearer ") else api_key
            if token != self._config.api_key:
                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid API key")

        @app.post("/run", response_model=RunResponse, responses={400: {"model": ErrorResponse}})
        async def run_endpoint(
            request: RunRequest,
            authorization: str | None = Header(None, alias="Authorization"),
        ) -> JSONResponse:
            """Execute the agent and return the response."""
            _validate_api_key(authorization)
            request_id = str(uuid.uuid4())
            start_time = time.time()
            self._metrics.record_request_start()

            try:
                if self._semaphore is None:
                    raise HTTPException(status_code=status.HTTP_503_SERVICE_UNAVAILABLE, detail="Server not ready")

                async with self._semaphore:
                    result = await self._run_agent(request)
                    duration_ms = (time.time() - start_time) * 1000
                    self._metrics.record_request_end(True, duration_ms)

                    if result.usage:
                        self._metrics.record_tokens(
                            result.usage.get("prompt_tokens", 0),
                            result.usage.get("completion_tokens", 0),
                        )

                    return JSONResponse(content=result.model_dump())

            except HTTPException:
                raise
            except ValidationError as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(False, duration_ms)
                return JSONResponse(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    content=ErrorResponse(
                        error="Invalid request",
                        code="VALIDATION_ERROR",
                        details={"errors": e.errors()},
                        request_id=request_id,
                    ).model_dump(),
                )
            except Exception as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(False, duration_ms)
                logger.error(f"Agent execution failed: {e}")
                return JSONResponse(
                    status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                    content=ErrorResponse(
                        error=str(e),
                        code="AGENT_EXECUTION_ERROR",
                        request_id=request_id,
                    ).model_dump(),
                )

        @app.post("/stream")
        async def stream_endpoint(
            request: RunRequest,
            authorization: str | None = Header(None, alias="Authorization"),
        ) -> EventSourceResponse:
            """Execute the agent and stream the response."""
            _validate_api_key(authorization)
            self._metrics.record_request_start()
            start_time = time.time()

            async def event_generator() -> AsyncGenerator[dict[str, Any], None]:
                try:
                    if self._semaphore is None:
                        yield {
                            "event": StreamEventType.ERROR.value,
                            "data": to_json(StreamEvent(type=StreamEventType.ERROR, data="Server not ready").model_dump()),
                        }
                        return

                    async with self._semaphore:
                        yield {
                            "event": StreamEventType.START.value,
                            "data": to_json(StreamEvent(type=StreamEventType.START).model_dump()),
                        }

                        async for event in self._stream_agent(request):

class ServerMetrics:
    """Thread-safe metrics collector."""

    def __init__(self) -> None:
        self._lock = Lock()
        self.requests_total = 0
        self.requests_active = 0
        self.requests_succeeded = 0
        self.requests_failed = 0
        self.total_response_time_ms = 0.0
        self.tokens_prompt = 0
        self.tokens_completion = 0
        self.start_time = time.time()
        self.request_ids_processed: list[str] = []
        self.last_request_id: str | None = None

    def record_request_start(self, request_id: str) -> None:
        with self._lock:
            self.requests_total += 1
            self.requests_active += 1
            self.last_request_id = request_id

    def record_request_end(self, request_id: str, success: bool, duration_ms: float) -> None:
        with self._lock:
            self.requests_active -= 1
            self.total_response_time_ms += duration_ms
            if success:
                self.requests_succeeded += 1
            else:
                self.requests_failed += 1
            # Keep last 100 request IDs for debugging
            self.request_ids_processed.append(request_id)
            if len(self.request_ids_processed) > 100:
                self.request_ids_processed.pop(0)

    def record_tokens(self, prompt: int, completion: int) -> None:
        with self._lock:
            self.tokens_prompt += prompt
            self.tokens_completion += completion

    def get_metrics(self) -> MetricsResponse:
        with self._lock:
            completed = self.requests_succeeded + self.requests_failed
            avg_time = self.total_response_time_ms / completed if completed > 0 else 0.0
            return MetricsResponse(
                requests_total=self.requests_total,
                requests_active=self.requests_active,
                requests_succeeded=self.requests_succeeded,
                requests_failed=self.requests_failed,
                avg_response_time_ms=avg_time,
                uptime_seconds=time.time() - self.start_time,
                tokens_used={
                    "prompt": self.tokens_prompt,
                    "completion": self.tokens_completion,
                    "total": self.tokens_prompt + self.tokens_completion,
                },
                last_request_id=self.last_request_id,
                recent_request_ids=self.request_ids_processed[-10:],
            )

class ServerMetrics:
    """Thread-safe metrics collector with request tracking."""

    def __init__(self) -> None:
        self._lock = Lock()
        self.requests_total = 0
        self.requests_active = 0
        self.requests_succeeded = 0
        self.requests_failed = 0
        self.total_response_time_ms = 0.0
        self.tokens_prompt = 0
        self.tokens_completion = 0
        self.start_time = time.time()
        self._recent_request_ids: list[str] = []
        self._max_recent_requests = 100

    def record_request_start(self, request_id: str) -> None:
        with self._lock:
            self.requests_total += 1
            self.requests_active += 1
            self._recent_request_ids.append(request_id)
            if len(self._recent_request_ids) > self._max_recent_requests:
                self._recent_request_ids.pop(0)

    def record_request_end(self, request_id: str, success: bool, duration_ms: float) -> None:
        with self._lock:
            self.requests_active -= 1
            self.total_response_time_ms += duration_ms
            if success:
                self.requests_succeeded += 1
            else:
                self.requests_failed += 1

    def record_tokens(self, prompt: int, completion: int) -> None:
        with self._lock:
            self.tokens_prompt += prompt
            self.tokens_completion += completion

    def get_metrics(self) -> MetricsResponse:
        with self._lock:
            completed = self.requests_succeeded + self.requests_failed
            avg_time = self.total_response_time_ms / completed if completed > 0 else 0.0
            return MetricsResponse(
                requests_total=self.requests_total,
                requests_active=self.requests_active,
                requests_succeeded=self.requests_succeeded,
                requests_failed=self.requests_failed,
                avg_response_time_ms=avg_time,
                uptime_seconds=time.time() - self.start_time,
                tokens_used={
                    "prompt": self.tokens_prompt,
                    "completion": self.tokens_completion,
                    "total": self.tokens_prompt + self.tokens_completion,
                },
                recent_request_ids=self._recent_request_ids.copy(),
            )


class AgentAPIServer:
    """Production-ready API server for BeeAI agents."""

    def __init__(
        self,

    @staticmethod
    def _generate_request_id() -> str:

    def _create_app(self) -> "FastAPI":

                "Required modules not found. Run 'pip install \"beeai-framework[deploy]\"' to install.",

                return

                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Missing API key")
            token = api_key.replace("Bearer ", "") if api_key.startswith("Bearer ") else api_key
            if token != self._config.api_key:
                raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid API key")

        @app.post("/run", response_model=RunResponse, responses={400: {"model": ErrorResponse}})
        async def run_endpoint(
            request: RunRequest,
            authorization: str | None = Header(None, alias="Authorization"),

            """Execute the agent and return the response."""
            _validate_api_key(authorization)
            request_id = self._generate_request_id()
            start_time = time.time()
            self._metrics.record_request_start(request_id)
            logger.info(f"[{request_id}] Starting /run request")

            try:
                if self._semaphore is None:
                    raise HTTPException(status_code=status.HTTP_503_SERVICE_UNAVAILABLE, detail="Server not ready")

                async with self._semaphore:
                    result = await self._run_agent(request, request_id)
                    duration_ms = (time.time() - start_time) * 1000
                    self._metrics.record_request_end(request_id, True, duration_ms)
                    logger.info(f"[{request_id}] Completed /run request in {duration_ms:.2f}ms")

                    if result.usage:
                        self._metrics.record_tokens(
                            result.usage.get("prompt_tokens", 0),
                            result.usage.get("completion_tokens", 0),
                        )

                    return JSONResponse(content=result.model_dump())

            except HTTPException:
                raise
            except ValidationError as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(request_id, False, duration_ms)
                logger.warning(f"[{request_id}] Validation error: {e}")
                return JSONResponse(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    content=ErrorResponse(
                        error="Invalid request",
                        code="VALIDATION_ERROR",
                        details={"errors": e.errors()},
                        request_id=request_id,
                    ).model_dump(),
                )
            except Exception as e:
                duration_ms = (time.time() - start_time) * 1000
                self._metrics.record_request_end(request_id, False, duration_ms)
                logger.error(f"[{request_id}] Agent execution failed: {e}")
                return JSONResponse(
                    status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                    content=ErrorResponse(
                        error=str(e),
                        code="AGENT_EXECUTION_ERROR",
                        request_id=request_id,

    async def _get_memory(self, session_id: str | None) -> BaseMemory:
        """Get or create memory for the session."""
        if session_id is None:
            memory = await self._agent.memory.clone()
            memory.reset()
            return memory

        if not await self._memory_manager.contains(session_id):
            memory = await self._agent.memory.clone()
            memory.reset()
            await self._memory_manager.set(session_id, memory)

        return await self._memory_manager.get(session_id)

    def _prepare_messages(self, request: RunRequest) -> list[UserMessage | AssistantMessage | SystemMessage]:
        """Convert request input to messages."""
        if isinstance(request.input, str):
            return [UserMessage(content=request.input)]

        messages: list[UserMessage | AssistantMessage | SystemMessage] = []
        for msg in request.input:
            if msg.role == MessageRole.USER:
                messages.append(UserMessage(content=msg.content))
            elif msg.role == MessageRole.ASSISTANT:
                messages.append(AssistantMessage(content=msg.content))
            elif msg.role == MessageRole.SYSTEM:
                messages.append(SystemMessage(content=msg.content))
        return messages

    async def _clone_agent_with_memory(self, session_id: str | None) -> AnyAgent:
        """Clone the agent with isolated memory for concurrent request safety."""
        cloned_agent = await self._agent.clone()
        cloned_agent.memory = await self._get_memory(session_id)
        return cloned_agent

    async def _run_agent(self, request: RunRequest, request_id: str) -> RunResponse:
        """Execute the agent and return the response."""
        messages = self._prepare_messages(request)

        run_kwargs: dict[str, Any] = {}
        if request.context:
            run_kwargs["context"] = request.context
        if request.max_iterations is not None:
            run_kwargs["max_iterations"] = request.max_iterations

        cloned_agent = await self._clone_agent_with_memory(request.session_id)
        try:
            logger.debug(f"[{request_id}] Executing agent with {len(messages)} message(s)")
            result = await cloned_agent.run(messages, **run_kwargs)

            usage = None
            if hasattr(result, "usage") and result.usage:
                usage = {
                    "prompt_tokens": result.usage.prompt_tokens,
                    "completion_tokens": result.usage.completion_tokens,
                    "total_tokens": result.usage.total_tokens,
                }

            response_messages = [
                MessageContent(
                    role=MessageRole.ASSISTANT if hasattr(msg, "role") and msg.role == "assistant" else MessageRole.USER,
                    content=msg.text,
                )
                for msg in result.output
            ]

            return RunResponse(
                output=result.last_message.text,
                messages=response_messages,
                session_id=request.session_id,
                usage=usage,
                request_id=request_id,
                metadata={
                    "iterations": len(result.iterations) if hasattr(result, "iterations") else None,
                },
            )
        except Exception as e:
            logger.error(f"[{request_id}] Agent execution error: {e}")
            raise AgentExecutionError(str(e), cause=e)

    async def _stream_agent(self, request: RunRequest, request_id: str) -> AsyncGenerator[StreamEvent, None]:
        """Execute the agent and stream events."""
        messages = self._prepare_messages(request)

        run_kwargs: dict[str, Any] = {}
        if request.context:
            run_kwargs["context"] = request.context
        if request.max_iterations is not None:
            run_kwargs["max_iterations"] = request.max_iterations

        cloned_agent = await self._clone_agent_with_memory(request.session_id)
        try:
            logger.debug(f"[{request_id}] Starting agent stream with {len(messages)} message(s)")
            run = cloned_agent.run(messages, **run_kwargs)

            final_output: str | None = None
            async for data, meta in run:
                event = self._convert_event(data, meta, request_id)
                if event:
                    yield event
                # Capture final answer from events
                if meta.name == "success" and hasattr(data, "data"):
                    final_output = data.data.text if hasattr(data.data, "text") else str(data.data)

            # If we didn't capture from success event, use a fallback
            if final_output is None:
                final_output = "Completed"

            yield StreamEvent(
                type=StreamEventType.DONE,
                data={
                    "output": final_output,
                    "session_id": request.session_id,
                },
                request_id=request_id,
            )

        except Exception as e:
            logger.error(f"[{request_id}] Stream error: {e}")
            yield StreamEvent(type=StreamEventType.ERROR, data=str(e), request_id=request_id)
            raise

    def _convert_event(self, data: Any, meta: EventMeta, request_id: str) -> StreamEvent | None:
        """Convert agent event to stream event."""
        if meta.name == "update" and hasattr(data, "update"):
            update = data.update
            if hasattr(update, "key"):
                if update.key == "thought":
                    return StreamEvent(
                        type=StreamEventType.THINKING,
                        data=str(update.value),
                        request_id=request_id,
                    )
                elif update.key == "tool_name":
                    return StreamEvent(
                        type=StreamEventType.TOOL_CALL,
                        data={"name": str(update.value)},
                        request_id=request_id,
                    )
                elif update.key == "tool_output":
                    return StreamEvent(
                        type=StreamEventType.TOOL_RESULT,
                        data={"output": str(update.value)},
                        request_id=request_id,
                    )
                elif update.key == "final_answer":
                    return StreamEvent(
                        type=StreamEventType.TOKEN,
                        data=str(update.value),
                        request_id=request_id,
                    )
        elif meta.name == "partial_update" and hasattr(data, "update"):
            update = data.update
            if hasattr(update, "key") and update.key == "final_answer":
                return StreamEvent(
                    type=StreamEventType.TOKEN,
                    data=str(update.value),
                    request_id=request_id,
                )

        return None
